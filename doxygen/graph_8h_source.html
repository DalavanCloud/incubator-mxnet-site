254 template <typename GNode, typename HashType,
255  typename FVisit, typename HashFunc,
256  typename InDegree, typename GetInput>
257 void PostOrderDFSVisit(const std::vector<GNode>& heads,
258  FVisit fvisit,
259  HashFunc hash,
260  InDegree indegree,
261  GetInput getinput) {
262  std::vector<std::pair<GNode, uint32_t> > stack;
263  std::unordered_set<HashType> visited;
264  for (auto& head : heads) {
265  HashType head_hash = hash(head);
266  if (visited.count(head_hash) == 0) {
267  stack.push_back(std::make_pair(head, 0));
268  visited.insert(head_hash);
269  }
270  while (!stack.empty()) {
271  std::pair<GNode, uint32_t>& back = stack.back();
272  if (back.second == indegree(back.first)) {
273  fvisit(back.first);
274  stack.pop_back();
275  } else {
276  const GNode& input = getinput(back.first, back.second++);
277  HashType input_hash = hash(input);
278  if (visited.count(input_hash) == 0) {
279  stack.push_back(std::make_pair(input, 0));
280  visited.insert(input_hash);
281  }
282  }
283  }
284  }

288 inline void DFSVisit(const std::vector<NodeEntry>& heads,
289  FVisit fvisit) {
290  typedef const NodePtr* GNode;
291  std::vector<GNode> head_nodes(heads.size());
292  std::transform(heads.begin(), heads.end(), head_nodes.begin(),
293  [](const NodeEntry& e)->GNode {
294  return &e.node;
295  });
296  PostOrderDFSVisit<GNode, Node*>(
297  head_nodes,
298  [fvisit](GNode n) { fvisit(*n); }, // FVisit
299  [](GNode n)->Node* { return n->get(); }, // HashFunc
300  [](GNode n)->uint32_t { // InDegree
301  if (!(*n)) return 0;
302  return (*n)->inputs.size() + (*n)->control_deps.size();
303  },
304  [](GNode n, uint32_t index)->GNode { // GetInput
305  if (index < (*n)->inputs.size()) {
306  return &(*n)->inputs.at(index).node;
307  } else {
308  return &(*n)->control_deps.at(index - (*n)->inputs.size());
309  }
310  });
