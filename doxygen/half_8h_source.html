186  Bits v, s;
187  v.f = value;
188  uint32_t sign = v.si & signN;
189  v.si ^= sign;
190  sign >>= shiftSign; // logical shift
191  s.si = mulN;
192  s.si = s.f * v.f; // correct subnormals
193  v.si ^= (s.si ^ v.si) & -(minN > v.si);
194  v.si ^= (infN ^ v.si) & -((infN > v.si) & (v.si > maxN));
