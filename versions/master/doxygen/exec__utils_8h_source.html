33 /*
34  * \brief setup default-storage tblobs from source NDArrays. If any source NDArray has non-default
35  * storage, it creates a temp NDArray with default storage and uses the temp tblob. The
36  * function also records the indices of non-default source NDArrays and the indices of
37  * their corresponding temporary NDArrays in the temp array.
38  * \param src list of source NDArray
39  * \param blobs list of tblobs to return
40  * \param temp_src list of source NDArrays which requires temporary default storage representation
41  * \param temp_dst list of temporary destination NDArrays for default storage representation
42  * \param idx_map mapping from indices in source NDArrays to indices in temp_dst. When not set,
43  indices are not recorded
44  * \return true if any source NDArray need to cast storage
45  */
46 inline bool SetupDefaultBlobsIn(const std::vector<NDArray>& src,
47  const std::vector<NDArray> *bufs,
48  std::vector<TBlob> *blobs,
49  std::vector<NDArray> *temp_src,
50  std::vector<NDArray> *temp_dst,
51  std::unordered_map<uint32_t, uint32_t> *idx_map) {
52  bool require_cast = false;
53  for (size_t i = 0; i < src.size(); i++) {
54  auto& nd = src[i];
55  bool is_default = nd.storage_type() == kDefaultStorage;
56 #if MXNET_USE_MKLDNN == 1
57  // We have to make sure it's default storage and default layout.
58  is_default = nd.IsDefaultData();
59 #endif
60  if (!is_default) {
61  (*idx_map)[i] = temp_dst->size();
62  NDArray temp = bufs != nullptr ? bufs->at(i) : NDArray(nd.shape(), nd.ctx(),
63  true, nd.dtype());
64 #if MXNET_USE_MKLDNN == 1
65  CHECK(temp.IsDefaultData());
66 #endif
67  temp_src->emplace_back(nd);
68  temp_dst->emplace_back(temp);
69  blobs->emplace_back(temp.data());
70  require_cast = true;
71  } else {
72  blobs->push_back(nd.data());
73  }
74  }
75  return require_cast;
76 }

112 /*
113  * \brief setup default-storage tblobs for input and output NDArrays.
114  * If any NDArray has non-default storage,
115  * it creates a temp NDArray with default storage and uses the temp tblob. The
116  * function also records the indices of non-default source NDArrays and the indices of
117  * their corresponding temporary NDArrays in the temp array.
118  */
119 inline void SetupDefaultBlobsInOut(const std::vector<NDArray> &ndinputs,
120  const std::vector<NDArray> &ndoutputs,
121  const std::vector<OpReqType> &req,
122  const std::vector<NDArray> *in_bufs,
123  const std::vector<NDArray> *out_bufs,
124  std::vector<TBlob> *input_blobs,
125  std::vector<TBlob> *output_blobs,
126  std::vector<NDArray> *pre_temp_src,
127  std::vector<NDArray> *pre_temp_dst,
128  std::vector<NDArray> *post_temp_src,
129  std::vector<NDArray> *post_temp_dst,
130  std::unordered_map<uint32_t, uint32_t> *in_temp_idx_map,
131  const std::vector<uint32_t> &mutate_idx) {
132  // populate input blobs
133  SetupDefaultBlobsIn(ndinputs, in_bufs, input_blobs, pre_temp_src, pre_temp_dst,
134  in_temp_idx_map);
135  // populate output blobs
136  SetupDefaultBlobsOut(ndoutputs, req, out_bufs, output_blobs, post_temp_dst,
137  post_temp_src);
138  // add mutable inputs to post temp list
139  for (const auto idx : mutate_idx) {
140  auto map_iter = in_temp_idx_map->find(idx);
141  if (map_iter != in_temp_idx_map->end()) {
142  post_temp_src->push_back(pre_temp_dst->at(map_iter->second));
143  post_temp_dst->push_back(ndinputs[idx]);
144  }
145  }
146 }

148 /*
149  * \brief cast the NDArrays in `src` and store the result in NDArrays in `dst`.
150  * This is only used for storage fallback in executor.
151  * \param src list of source NDArray to cast
152  * \param dst list of destionation NDArray which hold the result of cast_storage operation
153  * \param ctx operator context for cast_storage operation
154  */
155 inline void CastNonDefaultStorage(const std::vector<NDArray>& src,
156  const std::vector<NDArray>& dst,
157  const OpContext& ctx,
158  const bool is_gpu) {
159  CHECK_EQ(dst.size(), src.size());
160  for (size_t i = 0; i < src.size(); i++) {
161  if (is_gpu) {
162 #if MXNET_USE_CUDA
163  CastStorageDispatch<gpu>(ctx, src[i], dst[i]);
164 #else
165  LOG(FATAL) << MXNET_GPU_NOT_ENABLED_ERROR;
166 #endif
167  } else {
168  CastStorageDispatch<cpu>(ctx, src[i], dst[i]);
169  }
170  }
