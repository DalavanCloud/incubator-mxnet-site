293  template<typename Device, int dim, typename DType>
294  inline mshadow::Tensor<Device, dim, DType> FlatToKD(
295  mshadow::Stream<Device> *stream = NULL) const {
296  mshadow::Shape<dim> shape;
297  shape[0] = 1;
298  // Pad higher dimensions in case dim > ndim()
299  for (int i = 0; i < dim - ndim(); ++i) {
300  shape[i] = 1;
301  }
302  // Collapse higher dimensions in case dim < ndim()
303  for (int i = 0; i < ndim() - dim + 1; ++i) {
304  shape[0] *= shape_[i];
305  }
306  // Preserve lower dimensions.
307  for (int i = std::max(0, ndim() - dim + 1); i < ndim(); ++i) {
308  shape[i - ndim() + dim] = shape_[i];
309  }
310  return this->get_with_shape<Device, dim, DType>(shape, stream);
311  }

371  std::ostringstream os;
