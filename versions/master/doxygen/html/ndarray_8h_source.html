686  Chunk(NDArrayStorageType storage_type_, const TShape &storage_shape_, Context ctx_,
687  bool delay_alloc_, int dtype, const std::vector<int> &aux_types_,
688  const std::vector<TShape> &aux_shapes_)

773  inline void CheckAndAlloc(const TShape &shape, const std::vector<TShape> &aux_shapes,
774  int dtype) {
775  // calculate size, perform allocation
776  if (kRowSparseStorage == storage_type) {
777  // For row sparse, aux_shape indicates the number of rows to allocate
778  auto aux_shape = aux_shapes[rowsparse::kIdx];
779  CheckAndAllocAuxData(rowsparse::kIdx, aux_shape);
780  TShape storage_shape(shape);
781  storage_shape[0] = aux_shape[0];
782  CheckAndAllocData(storage_shape, dtype);
783  } else if (kCSRStorage == storage_type) {
784  CheckAndAllocAuxData(csr::kIndPtr, aux_shapes[csr::kIndPtr]);
785  CheckAndAllocAuxData(csr::kIdx, aux_shapes[csr::kIdx]);
786  CheckAndAllocData(aux_shapes[csr::kIdx], dtype);
787  } else {
788  LOG(FATAL) << "Storage type " << storage_type << " not implemented for CheckAndAlloc";
789  }
790  }
791  // create storage handle for data based on shape and dtype, assuming ctx is set
792  // storage shape is also updated
793  // if data is already allocated, try reuse the storage. Otherwise, free the current one
794  // and allocate new storage
795  inline void CheckAndAllocData(const TShape &shape, int dtype) {
796  CHECK_NE(aux_shapes.size(), 0) << "data is expected to be allocated after aux_data";
797  auto dbytes = shape.Size() * mshadow::mshadow_sizeof(dtype);
798  if (shandle.size < dbytes) {
799  // free storage if necessary and alloc again
800  if (shandle.size > 0) Storage::Get()->Free(shandle);
801  // init storage
802  shandle = Storage::Get()->Alloc(dbytes, ctx);
803  }
804  // init shape
805  storage_shape = shape;
806  // delay_alloc is only set when data storage handle is present
807  delay_alloc = false;
808  }
809  // create storage handle for aux data based on shape
810  // this function assumes ctx, aux shapes and aux types are set
811  // aux shape is also updated
812  // if aux data is already allocated, try reuse the storage. Otherwise, free the current one
813  // and allocate new storage
814  inline void CheckAndAllocAuxData(size_t i, const TShape &shape) {
815  CHECK_EQ(shape.ndim(), 1) << "shape must be 1D in CheckAndAllocAuxData";
816  CHECK_NE(storage_type, kUndefinedStorage)
817  << "storage type cannot be kUndefinedStorage in CheckAndAllocAuxData";
818  CHECK_NE(storage_type, kDefaultStorage)
819  << "storage type cannot be kDefaultStorage in CheckAndAllocAuxData";
820  if (aux_handles.size() <= i) {
821  aux_handles.resize(i + 1);
822  }
823  size_t aux_bytes = shape.Size() * mshadow::mshadow_sizeof(aux_types[i]);
824  if (aux_handles[i].size < aux_bytes) {
825  // free storage if necessary and alloc again
826  if (aux_handles[i].size > 0) Storage::Get()->Free(aux_handles[i]);
827  // init aux storage
828  aux_handles[i] = Storage::Get()->Alloc(aux_bytes, ctx);
829  }
830  // init shape
831  set_aux_shape(i, shape);
832  }
834  ~Chunk() {

836  Storage::Handle h = this->shandle;
837  std::vector<Storage::Handle> aux_h = this->aux_handles;
838  Engine::Get()->DeleteVariable([h, aux_h, skip_free](RunContext s) {
839  if (skip_free == false) {

842  if (aux_h[i].size > 0) Storage::Get()->Free(aux_h[i]);
843  }
844  }
845  }, shandle.ctx, var);
846  }
