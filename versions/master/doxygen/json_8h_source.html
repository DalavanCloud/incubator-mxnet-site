673 inline void JSONReader::ReadString(std::string *out_str) {
674  int ch = NextNonSpace();
675  CHECK_EQ(ch, '\"')
676  << "Error at" << line_info()
677  << ", Expect \'\"\' but get \'" << static_cast<char>(ch) << '\'';
678 #ifndef _LIBCPP_SGX_NO_IOSTREAMS
679  std::ostringstream output;
680 #else
681  std::string output = "";
682 #endif
683  while (true) {
684  ch = NextChar();
685  if (ch == '\\') {
686  char sch = static_cast<char>(NextChar());
687  switch (sch) {
688  case 'r': Extend(&output, "\r"); break;
689  case 'n': Extend(&output, "\n"); break;
690  case '\\': Extend(&output, "\\"); break;
691  case 't': Extend(&output, "\t"); break;
692  case '\"': Extend(&output, "\""); break;
693  default: LOG(FATAL) << "unknown string escape \\" << sch;
694  }
695  } else {
696  if (ch == '\"') break;
697  Extend(&output, static_cast<char>(ch));
698  }
699  if (ch == EOF || ch == '\r' || ch == '\n') {
700  LOG(FATAL)
701  << "Error at" << line_info()
702  << ", Expect \'\"\' but reach end of line ";
703  }
704  }
705 #ifndef _LIBCPP_SGX_NO_IOSTREAMS
706  *out_str = output.str();
707 #else
708  *out_str = output;
709 #endif
